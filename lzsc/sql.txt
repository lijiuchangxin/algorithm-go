CREATE TABLE `customer_alteration` (
  `id` int NOT NULL AUTO_INCREMENT,
  `customer_id` int NOT NULL DEFAULT '0',
  `user_id` int NOT NULL DEFAULT '0',
  `user_nick_name` varchar(20) NOT NULL DEFAULT '',
  `alter_time` int NOT NULL DEFAULT '0',
  `summary` varchar(200) NOT NULL DEFAULT '',
  `feed_type` varchar(20) NOT NULL DEFAULT 'alter',
  PRIMARY KEY (`id`),
  KEY `fk_customer_alteration_1_idx` (`customer_id`),
  CONSTRAINT `fk_customer_alteration_1` FOREIGN KEY (`customer_id`) REFERENCES `ut_customer` (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `customer_follow_up` (
  `id` int NOT NULL AUTO_INCREMENT,
  `create_at` int NOT NULL DEFAULT '0',
  `updated_at` int NOT NULL DEFAULT '0',
  `customer_id` int NOT NULL DEFAULT '0',
  `feed_type` varchar(20) NOT NULL DEFAULT 'follow up',
  `user_id` int NOT NULL DEFAULT '0',
  `user_avatar` varchar(200) NOT NULL DEFAULT '',
  `user_nick_name` varchar(20) NOT NULL DEFAULT '',
  PRIMARY KEY (`id`),
  KEY `fk_customer_follow_up_1_idx` (`customer_id`),
  CONSTRAINT `fk_customer_follow_up_1` FOREIGN KEY (`customer_id`) REFERENCES `ut_customer` (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
